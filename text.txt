SELECT [a].[Id], [a].[AccessFailedCount], [a].[AccountNumber], [a].[Address], [a].[CheckrCandidateId], [a].[CheckrStatus], [a].[CompanyId], [a].[ConcurrencyStamp], [a].[CreatedDate], [a].[DefaultActive], [a].[DefaultMeasurementSystem], [a].[DeletedDate], [a].[DriverStatus], [a].[Email], [a].[EmailConfirmed], [a].[FullName], [a].[Latitude], [a].[LockoutEnabled], [a].[LockoutEnd], [a].[Longitude], [a].[MobileRefreshToken], [a].[NormalizedEmail], [a].[NormalizedUserName], [a].[OldPassword], [a].[OldUserId], [a].[PasswordHash], [a].[PaymentMethod], [a].[PhoneNumber], [a].[PhoneNumberConfirmed], [a].[ReferralAccountId], [a].[ReferralCode], [a].[RefreshToken], [a].[SecurityStamp], [a].[StripeEntityId], [a].[Twillio], [a].[TwoFactorEnabled], [a].[UpdatedDate], [a].[UserName], [a].[UserStatus], [a].[ZenDeskId]
FROM [AspNetUsers] AS [a]
WHERE (([a].[DeletedDate] IS NULL) AND ([a].[UserStatus] = 0)) AND ((
    SELECT TOP(1) [a1].[Name]
    FROM [AspNetUserRoles] AS [a0]
    INNER JOIN [AspNetRoles] AS [a1] ON [a0].[RoleId] = [a1].[Id]
    WHERE [a].[Id] = [a0].[UserId]) = N'CompanyCustomer')
